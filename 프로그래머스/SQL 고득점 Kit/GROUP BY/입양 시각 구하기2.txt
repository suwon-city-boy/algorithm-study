SELECT HOURS.HOUR, NVL(OUTS.COUNT, 0) COUNT
  FROM (
           SELECT TO_CHAR(DATETIME, 'HH24') HOUR, COUNT(DATETIME) COUNT
             FROM ANIMAL_OUTS
            GROUP BY TO_CHAR(DATETIME, 'HH24')
           HAVING TO_CHAR(DATETIME, 'HH24') BETWEEN 0 AND 24
       ) OUTS,
       (
           SELECT LEVEL-1 HOUR
             FROM DUAL
          CONNECT BY LEVEL <= 24
       ) HOURS
 WHERE OUTS.HOUR(+) = HOURS.HOUR
 ORDER BY HOURS.HOUR
